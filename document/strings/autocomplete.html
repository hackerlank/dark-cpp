autocomplete 

#include <dark-cpp/strings/autocomplete.hpp>

autocomplete_t/wautocomplete_t 提供了 根據傳入命令 返回 完整 可選命令的 功能

使用 方式 

1	實例化 autocomplete_t/wautocomplete_t

2	調用 void get_cmds(const std::wstring& cmd,const std::vector<std::wstring>& cmds,std::vector<std::wstring>& outs,bool icase = false)			
	

3	從 outs 中 獲取 匹配的 命令

example 

#include <dark-cpp/strings/autocomplete.hpp>

int _tmain(int argc, _TCHAR* argv[])
{
	dark::strings::autocomplete_t autocomplete;
	std::vector<std::string> cmds;
	cmds.push_back("Create table");
	cmds.push_back("create index");

	std::string cmd = "create";
	std::vector<std::string> outs;
	autocomplete.get_cmds(cmd,cmds,outs,true);

	for(std::size_t i=0;i<outs.size();++i)
	{
		std::cout<<outs[i]<<"\n";
	}
	return 0;
}
